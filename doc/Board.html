    # File game.rb, line 119
def printBoard
        puts "Player 1's Board"
        @p1Board.each do |c|
                c.printCard
        end
        puts "Player 2's Board";
        @p2Board.each do |c|
                c.printCard
        end
        puts "Player 1's GY"
        @p1GY.each do |c|
                c.printCard
        end
        puts "Player 2's GY";
        @p2GY.each do |c|
                c.printCard
        end
end
    resetBoard(player) click to toggle source

    Refreshes the ability to attack for any card on the player's board. Should happen at the start of every turn

    # File game.rb, line 105
def resetBoard (player)
        if(player.side == 1)
                @p1Board.each do |c|
                        c.played = false
                end
        elsif(player.side == 2)
                @p2Board.each do |c|
                        c.played = false
                end
        end
end
