    # File game.rb, line 165
def die (board, player)
        @currentHealth = health
        if player.side == 1
                board.p1Board.delete(self)
                board.p1GY.push(self)
        elsif player.side == 2
                board.p2Board - [self]
                board.p2GY.push(self)
        end
end
